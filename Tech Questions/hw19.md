-- Create Keyspace
CREATE KEYSPACE chuwa 
WITH replication = {'class': 'SimpleStrategy', 'replication_factor': 1};

-- Create Table with multiple partition keys and clustering key
CREATE TABLE chuwa.trainee_information (
    trainee_id UUID,
    department TEXT,
    batch TEXT,
    join_date TIMESTAMP,
    name TEXT,
    email TEXT,
    phone TEXT,
    PRIMARY KEY ((trainee_id, department), batch, join_date)
);
